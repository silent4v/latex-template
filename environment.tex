% 自訂圖片插入指令（固定位置、置中、標題）
\newcommand{\InsertFigure}[3][]{%
  \begin{figure}[H]
    \centering
    \includegraphics[width=0.8\linewidth]{#2}
    \caption{#3}
    \ifx&#1&%
    \else
      \label{#1}
    \fi
  \end{figure}
}
